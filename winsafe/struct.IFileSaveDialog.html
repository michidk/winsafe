IFileSaveDialog COM interface over IFileSaveDialogVT.

Automatically calls IUnknown::Release when the object goes out of scope.

Examples

use winsafe::prelude::*;
use winsafe::{co, CoCreateInstance, IFileSaveDialog};

let obj = CoCreateInstance::<IFileSaveDialog>(
    &co::CLSID::FileSaveDialog,
    None,
    co::CLSCTX::INPROC_SERVER,
)?;
