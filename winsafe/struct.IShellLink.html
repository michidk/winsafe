pub struct IShellLink(_);

Expand description

IShellLinkW COM interface over IShellLinkVT.

Automatically calls Release when the object goes out of scope.

Examples

use winsafe::prelude::*;
use winsafe::{co, CoCreateInstance, IShellLink};

let obj = CoCreateInstance::<IShellLink>(
    &co::CLSID::ShellLink,
    None,
    co::CLSCTX::INPROC_SERVER,
)?;
