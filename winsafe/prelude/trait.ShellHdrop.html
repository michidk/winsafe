fn iter<'a>(
    &'a self
) -> WinResult<Box<dyn Iterator<Item = WinResult<String>> + 'a>>

Returns an iterator over the dropped files by calling HDROP::DragQueryFile consecutively, then frees the handle by calling HDROP::DragFinish.

Examples

Iterating over the strings:

use winsafe::prelude::*;
use winsafe::HDROP;

let hdrop: HDROP; // initialized somewhere

for file_path in hdrop.iter()? {
    let file_path = file_path?;
    println!("File: {}", file_path);
}

Collecting the strings into a Vec:

use winsafe::prelude::*;
use winsafe::{HDROP, WinResult};

let hdrop: HDROP; // initialized somewhere

let file_paths = hdrop.iter()?
    .collect::<WinResult<Vec<_>>>()?;
